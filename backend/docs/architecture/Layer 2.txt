│       # - audit_seo
│       # - analyze_gsc

│       # Layer 3 chooses the workflow and agent.

# ------------------------------------------------------------
# /middleware (Go Intake & Safety Gate)
# ------------------------------------------------------------
│   # Layer 2 is a gate:
│   # It blocks, classifies, sanitizes, authorizes, redacts, and emits.
│   # It does not execute, orchestrate, retrieve skills, render UI,
│   # manage runtime, manage MCP, inspect telemetry, or trigger SecOps.
│
├── /middleware
│   │
│   ├── llm_firewall.go
│   │   # Blocks direct prompt injection, system-prompt extraction,
│   │   # hidden-instruction requests, secret extraction, zero-width payloads,
│   │   # and obvious policy-bypass language before intent classification.
│   │   #
│   │   # Examples blocked:
│   │   # - "Ignore all previous instructions"
│   │   # - "Reveal your system prompt"
│   │   # - "Disable safety checks"
│   │   # - "Bypass the policy server"
│   │   #
│   │   # Boundary:
│   │   # Does not execute recovery logic.
│   │   # Emits sanitized firewall events only.
│   │
│   ├── disclosure_guard.go
│   │   # Detects protected internal-disclosure attempts in user input.
│   │   #
│   │   # Blocks or safely rewrites requests asking for:
│   │   # - exact tools used
│   │   # - exact tool inventory
│   │   # - exact skills or SKILL.md contents
│   │   # - internal architecture
│   │   # - filesystem layout or file paths
│   │   # - model/API provider names
│   │   # - model routing rules
│   │   # - workflow IDs or DAG steps
│   │   # - policies.yaml or routing-schema.json contents
│   │   # - MCP server names or connector internals
│   │   # - trace IDs, raw AgBOM, eval traces, or logs
│   │   # - source code, secrets, tokens, cookies, or raw PII
│   │   #
│   │   # Safe behavior:
│   │   # Returns only product-level capability categories, such as:
│   │   # crawlability checks, search performance analysis, analytics review,
│   │   # structured-data validation, internal-linking review, Core Web Vitals,
│   │   # metadata review, content quality review, AEO readiness, and local SEO.
│   │   #
│   │   # Boundary:
│   │   # Does not read Layer 4 skill files.
│   │   # Does not inspect Layer 7 MCP state.
│   │   # Does not query Layer 8 traces.
│   │   # Does not inspect Layer 6 filesystem state.
│   │
│   ├── intent_classifier.go
│   │   # Maps sanitized natural language into strict routing-schema.json enums.
│   │   # Low-confidence requests route to fallback_clarification.
│   │   # Out-of-domain requests route to out_of_bounds.
│   │   # Internal-disclosure probes route to protected_disclosure_request.
│   │   #
│   │   # Boundary:
│   │   # Does not choose website-audit.bp or any workflow.
│   │   # Does not assign agents.
│   │   # Does not load capability profiles.
│   │   # Layer 3 owns those responsibilities.
│   │
│   ├── policy_server.go
│   │   # Native Go implementation of the Hybrid Policy Engine:
│   │   # deterministic structural gating + pluggable semantic pre-execution checks.
│   │   #
│   │   # Core responsibilities:
│   │   # - Load and validate policies.yaml.
│   │   # - IsToolAllowed(): deterministic allow/deny by tool, env, role, intent.
│   │   # - AuthorizeTool(): fail-closed authorization before runtime execution.
│   │   # - CheckActionSemantic(): semantic check for PII leakage, secret exposure,
│   │   #   prompt-injection propagation, unsafe action payloads, policy bypass,
│   │   #   and protected metadata disclosure.
│   │   #
│   │   # Boundary:
│   │   # Verifies approval metadata only.
│   │   # Does not initiate MFA, approval UI, or human-review workflows.
│   │   # Does not mint, revoke, or refresh JIT credentials.
│   │   # Does not enforce runtime sandbox controls.
│   │
│   ├── context_resolver.go
│   │   # High-speed regex translation engine executing Context Hygiene.
│   │   # Resolves only trusted [[VARIABLE_NAME]] placeholders from the provided
│   │   # runtime state map.
│   │   #
│   │   # Rule:
│   │   # Missing variables remain intact.
│   │   # Example:
│   │   # "Audit [[TARGET_URL]] for [[UNKNOWN_VAR]]"
│   │   # -> "Audit livesoccer24.com for [[UNKNOWN_VAR]]"
│   │   #
│   │   # Boundary:
│   │   # Does not fetch missing values from memory.
│   │   # Does not query Layer 3, 4, 6, 7, or 8 for missing context.
│   │   # Does not guess or hallucinate values.
│   │
│   ├── tool_policy_engine.go
│   │   # Framework-level interceptor pipeline for proposed tool calls.
│   │   #
│   │   # Required sequence:
│   │   # 1. Validate tool name.
│   │   # 2. Validate parameter shape.
│   │   # 3. Sanitize parameters.
│   │   # 4. Serialize parameters.
│   │   # 5. Authorize via policies.yaml.
│   │   # 6. Run semantic pre-execution safety check.
│   │   # 7. Emit sanitized allow/block event.
│   │   # 8. Allow only verified payloads to proceed toward Layer 6.
│   │   #
│   │   # Fail-closed:
│   │   # Unknown, malformed, unserializable, over-limit, or unsafe payloads stop here.
│   │   #
│   │   # Boundary:
│   │   # Does not execute tools.
│   │   # Does not connect to MCP servers.
│   │   # Does not open network sockets.
│   │   # Does not inspect runtime sandbox state.
│   │   # Does not enforce filesystem mounts.
│   │
│   └── outbound_response_guard.go
│       # Validates assistant-visible text responses before release to the user.
│       #
│       # Prevents accidental leakage of:
│       # - exact internal tool names
│       # - exact skill names
│       # - internal architecture trees
│       # - file paths and filesystem layout
│       # - API/model provider names
│       # - routing rules or workflow IDs
│       # - MCP server names
│       # - policies.yaml / routing-schema.json contents
│       # - source code
│       # - raw logs, trace IDs, AgBOM, eval traces
│       # - hidden prompts, secrets, tokens, cookies, raw PII
│       #
│       # If protected metadata is detected, replace with a safe
│       # product-level capability explanation.
│       #
│       # Boundary:
│       # Does not render A2UI.
│       # Does not create UI cards.
│       # Does not query Layer 8 traces.
│       # Does not query Layer 4 skill files.
│       # Does not query Layer 7 MCP state.
│       # Does not modify audit outputs directly.

# ============================================================
# LAYER 2 REQUEST FLOW
# ============================================================

user_request_flow {

  01_raw_user_input
    -> NormalizeInput

  02_firewall
    -> InspectInput
    -> block if prompt injection, secret extraction, or policy bypass is detected

  03_disclosure_guard
    -> InspectDisclosureRequest
    -> if protected internal-disclosure request:
         emit protected_disclosure_blocked
         return safe product-level capability response policy
         do not expose internal metadata

  04_intent_classification
    -> ClassifyIntent
    -> ValidateIntentAgainstSchema
    -> FallbackIfLowConfidence

  05_context_hygiene
    -> ResolveVariables using trusted state map only
    -> leave unresolved placeholders intact

  06_emit_event
    -> EmitIntakeEvent(intent_classified OR intent_fallback OR intent_out_of_bounds)

  07_handoff
    -> pass only validated intent enum and sanitized context to Layer 3
    -> do not choose workflow
    -> do not execute tool
    -> do not load skills
}


# ============================================================
# LAYER 2 TOOL-CALL INTERCEPTION FLOW
# ============================================================

tool_call_interception_flow {

  01_proposed_tool_call_from_layer_3
    -> ValidateParamShape

  02_sanitize
    -> SanitizeParams
    -> SerializeParams

  03_policy
    -> AuthorizeTool using policies.yaml

  04_semantic_check
    -> CheckActionSemantic on serialized action payload

  05_emit_event
    -> EmitIntakeEvent(tool_policy_allowed OR tool_policy_blocked OR semantic_policy_blocked)

  06_handoff
    -> allow only verified payload to proceed toward Layer 6 runtime
    -> Layer 2 does not execute the tool
}


# ============================================================
# LAYER 2 OUTBOUND RESPONSE FLOW
# ============================================================

outbound_response_flow {

  01_draft_assistant_text
    -> ValidateOutboundResponse

  02_redaction
    -> if protected metadata detected:
         RedactProtectedMetadata
         EmitIntakeEvent(outbound_response_redacted)

  03_release
    -> release safe text response
    -> Layer 2 does not render A2UI or modify UI state
}
