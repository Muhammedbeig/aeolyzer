# 02. WHAT IS INSIDE EACH INSTRUCTION SET
# ------------------------------------------------------------

research.skill {
  search competitor content before asking questions
  search industry news for timely hooks
  search user site to avoid cannibalization
  use webSearch() for fast lookups
  use deepResearch() for evidence-heavy pieces
  use scrapePage() to analyze specific URLs
  collect source URLs for every stat and quote
}

brief-building.skill {
  topic    : what are we writing about
  angle    : unique perspective or thesis
  audience : who is reading this
  intent   : what should reader know feel or do
  length   : target word count
  cta      : where should reader go next
  save via updateBrief() before writing begins
}

topic-discovery.skill {
  step_01 : call quickContext() for brand and competitors
  step_02 : search competitor blogs
  step_03 : search industry news
  step_04 : find content gaps
  step_05 : suggest 2-3 strategic topics with hook and value
  never suggest top 10 or future of Y topics
}

title-generation.skill {
  identify surface intent   : what query literally asks
  identify emotional intent : what reader feels but did not type
  identify business intent  : what justifies reader time or budget
  apply pattern             : [surface answer] + [emotional resolution]
  generate specific useful differentiated titles only
}

outline-structure.skill {
  title = H1 via generateTitle()
  intro : no H2, flows as prose under title
  H2    : major sections each advancing thesis
  H3    : subsections for depth or comparison
  every section must have at least 2 non-paragraph elements
  end with conclusion that resolves thesis and leads to CTA
}

writing.skill {
  write in clear authoritative evidence-backed style
  take a position, do not just summarize
  explain why every major fact matters
  use direct analytical language
  replace vague claims with specific numbers
  keep paragraph length controlled
  match technical depth to audience knowledge level
}

seo-search-intent.skill {
  identify primary keyword
  align content type to intent
  check cannibalization risk
  cover adjacent questions
  use question-based subheadings where useful
  add concise self-contained answers for AI extraction
}

inline-linking.skill {
  minimum 3 links per 1000 words
  internal links to user site pages
  external links to authority non-competitor domains only
  descriptive anchor text always
  never link to blocked competitor domains
}

brand-safety.skill {
  never cite or link to competitor domains
  never elevate competitors as hero case studies
  frame analysis to strengthen user authority
  keep messaging aligned with brand voice
  anonymize competitor data when useful
}

formatting-rules.skill {
  never use em dash
  never use emojis or unicode symbols
  never write more than 2 consecutive plain paragraphs
  every H2 section must have at least 2 non-paragraph elements
  H2 sections over 200 words must have at least 1 H3
  use bullet lists, tables, blockquotes, callouts, H3s
}

post-write-checklist.skill {
  every planned section is present
  thesis is clear and consistent
  word count is close to target
  minimum link count is met
  no text walls
  no duplicate H2 headings
  no em dashes
  SEO settings can be generated from finished draft
}

seo-outputs.skill {
  meta title
  meta description
  url slug
  og title
  og description
  faqs (3-5 questions and answers)
  schema markup JSON-LD
  internal link suggestions
}

hidden-intent-analysis.skill {
  surface intent   : what the query literally asks
  emotional intent : what the reader feels but did not type
  business intent  : what justifies time or budget
  golden thread    : single narrative resolving all three layers
  runs from title through intro through body to conclusion
}

content-craft.skill {
  AEO: sections lead with 40-60 word direct answer
  evidence density: at least 1 data point per major section
  readability: paragraphs 40-60 words, 2-3 sentences
  UGC intelligence: use real user language from forums and reddit
  self-contained sections: each H2 readable in isolation
}

anti-redundancy.skill {
  never make the same point in two different sections
  each section must advance the argument
  core thesis stated once in introduction only
  before writing each section check what was already said
}

editorial-voice.skill {
  take positions, do not just present information
  interpret facts, explain what they mean
  use direct analytical language
  never hedge more than once per paragraph
  distinguish from press release: analyze not just inform
}

depth-evidence.skill {
  go beyond the press release
  address what is missing from standard narrative
  reference specific numbers, dates, companies, quotes
  include counter-arguments
  weave data into prose, do not list facts
}

sources-intelligence.skill {
  call getSourcesInsights() early alongside quickContext()
  competitor blocklist: never link to these domains
  authority sources: safe outbound link targets
  trending content types: inform format suggestions
}

strategic-intelligence.skill {
  competitive gaps : competitor covers it, you do not
  timing gaps      : recent news, nobody covered it yet
  depth gaps       : existing coverage is shallow
  audience gaps    : content exists for wrong reader
  every suggestion needs hook and strategic value
}

memory-system.skill {
  memory.md : brand identity, competitors, positioning, learned prefs
  tone.md   : writing voice, style rules, banned phrases
  readMemoryDoc()        : re-read docs when needed
  proposeMemoryUpdate()  : propose changes, never update silently
}

optimize-mode.skill {
  seo-inspect      : read current seo state, never overwrite existing
  intent-detection : classify search intent from content
  metadata         : meta title, description, slug, og fields
  faq-generation   : 3-5 real questions with direct answers
  schema           : JSON-LD, type matched to content
  internal-linking : find related site pages, suggest anchor text
  seo-validation   : check all fields align with content
}

# ------------------------------------------------------------
# 03. ROUTING / EXECUTION LOGIC
# ------------------------------------------------------------

# how the system decides what to do next
# inferred from exposed behavior, not raw router source code

routing.logic {

  step_01 : read session context
            datetime, project, memory, tone, content_type,
            canvas_state, seo_settings, brief

  step_02 : read user message

  step_03 : classify intent
            write content
            plan content
            research topic
            answer question
            optimize SEO
            audit page
            find gaps
            generate task list
            generate llms.txt
            edit existing content
            switch content type
            update memory

  step_04 : select skill groups needed for task

  step_05 : call required tools in sequence
            quickContext() first if brand context needed
            getSourcesInsights() early for competitor blocklist
            webSearch() or deepResearch() for research tasks
            getSitePages() for internal linking
            scrapePage() for URL analysis
            writeSection() for canvas output
            proposeEdit() for surgical canvas edits
            updateBrief() before writing begins
            generateTitle() as soon as topic is known
            proposeMemoryUpdate() when new rule is learned

  step_06 : output to correct channel
            canvas : all content, files, architecture docs
            chat   : 1-2 sentence coordination only
            brief  : saved via updateBrief()

}

# ------------------------------------------------------------
# 05. INTERNAL WORKFLOW BLUEPRINTS
# ------------------------------------------------------------

workflow_01 : write-mode-content {
  step_01 : call quickContext() and getSourcesInsights()
  step_02 : research topic if needed via webSearch()
  step_03 : ask 1-2 batched questions via askQuestion()
  step_04 : call generateTitle() as soon as topic is known
  step_05 : call updateBrief() before writing
  step_06 : write section by section via writeSection()
  step_07 : track running word count from returned wordCount
  step_08 : stop when target word count is reached
  step_09 : run post-write checklist
  step_10 : suggest SEO settings generation
}

workflow_02 : seo-optimize {
  step_01 : read current seo_settings
  step_02 : identify null or empty fields
  step_03 : read article title and content
  step_04 : classify search intent
  step_05 : generate metadata for null fields only
  step_06 : generate FAQs from article content
  step_07 : generate schema JSON-LD
  step_08 : find internal link candidates via getSitePages()
  step_09 : validate all generated fields
  step_10 : never overwrite existing values
}

workflow_03 : content-gap-finding {
  step_01 : call quickContext() for brand and competitors
  step_02 : call getSourcesInsights() for blocklist
  step_03 : search competitor content via webSearch()
  step_04 : inspect user site via getSitePages()
  step_05 : identify gap types
  step_06 : output prioritized gap list with angles
}

workflow_04 : homepage-audit {
  step_01 : scrape homepage via scrapePage()
  step_02 : inspect headings, copy, CTAs, trust signals
  step_03 : check five core homepage questions
  step_04 : compare against memory.md positioning
  step_05 : output audit report
}

workflow_05 : memory-update {
  step_01 : identify new brand or tone rule from conversation
  step_02 : check if rule already exists in memory or tone doc
  step_03 : call proposeMemoryUpdate() with proposed content
  step_04 : wait for user accept or reject
  step_05 : never update silently
}
